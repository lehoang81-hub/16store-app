'use client';

import { useState } from 'react';
import { HeritageListing } from '@/components/HeritageListing';
import { useRouter } from 'next/navigation';
import {
  initiateLoan,
  confirmLoan,
  returnLoan,
  escalateLoan,
} from '@/lib/actions/loan-actions';

interface Props {
  passportId: string;
  qrCode: string;
  assetId: string;
  objectType: string;
  brand: string;
  model: string;
  isOwner: boolean;
  isLost: boolean;
  isBorrowing?: boolean;   // User đang mượn vật phẩm này
  activeLoanId?: string;   // Loan ID nếu đang active
  identityStatus: string;
  securityTier: string;
}

type Panel = 'none' | 'loan' | 'lost' | 'transfer' | 'journal';

export function PassportActions({
  passportId, qrCode, assetId, objectType,
  brand, model, isOwner, isLost,
  isBorrowing, activeLoanId,
  identityStatus, securityTier,
}: Props) {
  const router = useRouter();
  const [panel, setPanel] = useState<Panel>('none');
  const [loading, setLoading] = useState(false);
  const [toast, setToast] = useState<{ type: string; title: string; message: string } | null>(null);

  // Loan form state
  const [borrowerHandle, setBorrowerHandle] = useState('');
  const [weeks, setWeeks] = useState(4);
  const [weeklyRate, setWeeklyRate] = useState(15);
  const [loanNote, setLoanNote] = useState('');

  const objectLabel = objectType === 'sneaker' ? 'giày' : objectType === 'watch' ? 'đồng hồ' : objectType === 'bag' ? 'túi' : 'vật phẩm';

  function showToast(t: typeof toast) {
    setToast(t);
    setTimeout(() => setToast(null), 5000);
  }

  // ── Loan handlers ─────────────────────────────────────────

  async function handleInitiateLoan() {
    if (!borrowerHandle.trim()) return;
    setLoading(true);
    const result = await initiateLoan({
      assetId,
      borrowerHandle,
      weeks,
      weeklyHlrRate: weeklyRate,
      loanNote: loanNote || undefined,
    });
    setLoading(false);
    showToast(result.toast);
    if (result.success) { setPanel('none'); router.refresh(); }
  }

  async function handleConfirmLoan() {
    if (!activeLoanId) return;
    setLoading(true);
    let lat: number | undefined, lng: number | undefined, acc: number | undefined;
    try {
      const pos = await new Promise<GeolocationPosition>((res, rej) =>
        navigator.geolocation.getCurrentPosition(res, rej, { enableHighAccuracy: true })
      );
      lat = pos.coords.latitude;
      lng = pos.coords.longitude;
      acc = Math.round(pos.coords.accuracy);
    } catch {}
    const result = await confirmLoan({ loanId: activeLoanId, lat, lng, accuracyM: acc });
    setLoading(false);
    showToast(result.toast);
    if (result.success) router.refresh();
  }

  async function handleReturnLoan() {
    if (!activeLoanId) return;
    setLoading(true);
    let lat: number | undefined, lng: number | undefined, acc: number | undefined;
    try {
      const pos = await new Promise<GeolocationPosition>((res, rej) =>
        navigator.geolocation.getCurrentPosition(res, rej, { enableHighAccuracy: true })
      );
      lat = pos.coords.latitude;
      lng = pos.coords.longitude;
      acc = Math.round(pos.coords.accuracy);
    } catch {}
    const result = await returnLoan({ loanId: activeLoanId, lat, lng, accuracyM: acc });
    setLoading(false);
    showToast(result.toast);
    if (result.success) router.refresh();
  }

  async function handleEscalate(action: 'extend' | 'report_lost' | 'freeze') {
    if (!activeLoanId) return;
    const confirm = window.confirm(
      action === 'freeze' ? '⚠️ Bạn chắc chắn muốn đóng băng tài khoản người mượn?' :
      action === 'report_lost' ? '🚨 Báo mất sẽ trừ 20 điểm uy tín của người mượn. Tiếp tục?' :
      'Gia hạn thêm 1 tuần?'
    );
    if (!confirm) return;
    setLoading(true);
    const result = await escalateLoan({ loanId: activeLoanId, action });
    setLoading(false);
    showToast(result.toast);
    if (result.success) router.refresh();
  }

  return (
    <div className="border border-line mb-10">

      {/* Toast */}
      {toast && (
        <div className={`fixed top-4 right-4 z-50 max-w-sm p-4 border font-mono text-sm
          ${toast.type === 'success' ? 'border-[#6ec070] bg-[#6ec070]/10 text-[#6ec070]' :
            toast.type === 'warning' ? 'border-hazard bg-hazard/10 text-hazard' :
            'border-rust bg-rust/10 text-rust'}`}
        >
          <div className="font-bold text-[11px] tracking-[0.14em] uppercase mb-1">{toast.title}</div>
          <div className="text-[11px] leading-[1.5] whitespace-pre-line">{toast.message}</div>
        </div>
      )}

      {/* Action buttons */}
      <div className="p-5 flex flex-wrap gap-3">

        {/* Owner actions */}
        {isOwner && !isLost && !isBorrowing && (
          <>
            <ActionBtn
              label="Cho mượn"
              icon="🤝"
              active={panel === 'loan'}
              onClick={() => setPanel(panel === 'loan' ? 'none' : 'loan')}
            />
            <ActionBtn
              label="Chuyển nhượng"
              icon="🔄"
              active={panel === 'transfer'}
              onClick={() => setPanel(panel === 'transfer' ? 'none' : 'transfer')}
            />
            <ActionBtn
              label="Ký gửi bán"
              icon="🏷️"
              onClick={() => panel === 'listing' && (  
<HeritageListing
    assetId={assetId}
    assetData={...}
    userHlr={currentUserHlr}
    isFirstAsset={...}
    onClose={() => setPanel('none')}
  />
)}
            />
            <ActionBtn
              label="Báo mất"
              icon="🚨"
              danger
              active={panel === 'lost'}
              onClick={() => setPanel(panel === 'lost' ? 'none' : 'lost')}
            />
          </>
        )}

        {/* Owner + active loan */}
        {isOwner && activeLoanId && (
          <>
            <ActionBtn label="Gia hạn" icon="📅" onClick={() => handleEscalate('extend')} loading={loading} />
            <ActionBtn label="Báo mất" icon="🚨" danger onClick={() => handleEscalate('report_lost')} loading={loading} />
            <ActionBtn label="Đóng băng" icon="🔒" danger onClick={() => handleEscalate('freeze')} loading={loading} />
          </>
        )}

        {/* Borrower actions */}
        {isBorrowing && activeLoanId && (
          <ActionBtn
            label="Xác nhận trả đồ"
            icon="✅"
            onClick={handleReturnLoan}
            loading={loading}
          />
        )}

        {/* Pending loan — borrower chưa confirm */}
        {!isOwner && activeLoanId && !isBorrowing && (
          <ActionBtn
            label="Xác nhận nhận đồ"
            icon="📦"
            onClick={handleConfirmLoan}
            loading={loading}
          />
        )}

        {/* Nhật ký — always */}
        <ActionBtn
          label="Nhật ký"
          icon="📝"
          active={panel === 'journal'}
          onClick={() => setPanel(panel === 'journal' ? 'none' : 'journal')}
        />
      </div>

      {/* ── Panel: Cho mượn ──────────────────────────────────── */}
      {panel === 'loan' && (
        <div className="border-t border-line p-5 space-y-4">
          <div className="font-mono text-[10px] text-rust tracking-[0.2em] uppercase flex items-center gap-2 before:content-[''] before:w-2 before:h-2 before:bg-rust">
            Cho mượn {objectLabel}
          </div>

          <div className="grid grid-cols-2 gap-4 max-sm:grid-cols-1">
            <div>
              <label className="font-mono text-[9px] text-concrete tracking-[0.15em] uppercase block mb-1">
                @Handle người mượn *
              </label>
              <input
                type="text"
                placeholder="@tranngoc"
                value={borrowerHandle}
                onChange={e => setBorrowerHandle(e.target.value)}
                className="w-full bg-ink border border-line text-bone text-sm p-2 font-mono focus:outline-none focus:border-bone-2"
              />
            </div>
            <div>
              <label className="font-mono text-[9px] text-concrete tracking-[0.15em] uppercase block mb-1">
                Số tuần *
              </label>
              <div className="flex gap-2">
                {[1, 2, 4, 8, 12].map(w => (
                  <button
                    key={w}
                    onClick={() => setWeeks(w)}
                    className={`flex-1 py-2 font-mono text-[10px] tracking-[0.1em] border transition-colors
                      ${weeks === w ? 'bg-rust/10 border-rust text-rust' : 'border-line text-concrete hover:border-bone-2'}`}
                  >
                    {w}W
                  </button>
                ))}
              </div>
            </div>
          </div>

          <div>
            <label className="font-mono text-[9px] text-concrete tracking-[0.15em] uppercase block mb-1">
              HLR Rate / tuần * <span className="text-concrete/50">(tham số động)</span>
            </label>
            <div className="flex items-center gap-4">
              <input
                type="range" min="1" max="100" value={weeklyRate}
                onChange={e => setWeeklyRate(Number(e.target.value))}
                className="flex-1"
              />
              <div className="font-display text-2xl text-rust min-w-[80px] text-right">
                {weeklyRate} HLR
              </div>
            </div>
            <div className="font-mono text-[9px] text-concrete mt-1">
              Bạn nhận: <span className="text-[#6ec070]">{Math.round(weeklyRate * 0.53)} HLR/tuần</span> ·
              Platform: <span className="text-concrete/70">{weeklyRate - Math.round(weeklyRate * 0.53)} HLR/tuần</span> ·
              Tổng ước tính: <span className="text-bone">{weeklyRate * weeks} HLR</span>
            </div>
          </div>

          <div>
            <label className="font-mono text-[9px] text-concrete tracking-[0.15em] uppercase block mb-1">
              Điều kiện bảo quản
            </label>
            <textarea
              value={loanNote}
              onChange={e => setLoanNote(e.target.value)}
              placeholder={`VD: Không mang ${objectLabel} đi xa quá 50km, bảo quản trong hộp...`}
              rows={2}
              className="w-full bg-ink border border-line text-bone text-sm p-2 font-mono resize-none focus:outline-none focus:border-bone-2"
            />
          </div>

          <div className="flex gap-3">
            <button
              onClick={handleInitiateLoan}
              disabled={loading || !borrowerHandle.trim()}
              className="flex-1 bg-rust text-ink py-3 font-mono text-[11px] font-bold tracking-[0.18em] uppercase hover:bg-bone transition-colors disabled:opacity-40"
            >
              {loading ? 'Đang gửi...' : '✦ Gửi gắm — Khởi động Kindness'}
            </button>
            <button
              onClick={() => setPanel('none')}
              className="border border-line text-concrete px-4 py-3 font-mono text-[10px] tracking-[0.14em] uppercase hover:text-bone transition-colors"
            >
              Huỷ
            </button>
          </div>
        </div>
      )}

      {/* ── Panel: Báo mất ───────────────────────────────────── */}
      {panel === 'lost' && (
        <div className="border-t border-rust/30 bg-rust/5 p-5 space-y-4">
          <div className="font-mono text-[10px] text-rust tracking-[0.2em] uppercase flex items-center gap-2 before:content-[''] before:w-2 before:h-2 before:bg-rust">
            Báo mất {objectLabel}
          </div>
          <p className="font-mono text-[11px] text-bone-2 leading-[1.6]">
            Sau khi báo mất:
            <br />• Mọi giao dịch chuyển nhượng bị khoá ngay
            <br />• Hệ thống phát cảnh báo toàn mạng
            <br />• Ai scan QR → bạn nhận thông báo vị trí
            <br />• Người tìm thấy nhận thưởng HLR
          </p>
          <div className="flex gap-3">
            <button
              onClick={() => router.push(`/passport/${qrCode}?action=report_lost`)}
              className="flex-1 bg-rust text-ink py-3 font-mono text-[11px] font-bold tracking-[0.18em] uppercase hover:bg-bone transition-colors"
            >
              🚨 Xác nhận báo mất
            </button>
            <button
              onClick={() => setPanel('none')}
              className="border border-line text-concrete px-4 py-3 font-mono text-[10px] uppercase hover:text-bone transition-colors"
            >
              Huỷ
            </button>
          </div>
        </div>
      )}

      {/* ── Panel: Nhật ký ───────────────────────────────────── */}
      {panel === 'journal' && (
        <div className="border-t border-line p-5">
          <JournalPanel
            passportId={passportId}
            objectLabel={objectLabel}
            isOwner={isOwner}
          />
        </div>
      )}

    </div>
  );
}

// ── Journal Panel ─────────────────────────────────────────────

function JournalPanel({
  passportId, objectLabel, isOwner,
}: {
  passportId: string; objectLabel: string; isOwner: boolean;
}) {
  const [search, setSearch] = useState('');
  const [filter, setFilter] = useState('all');
  const [adding, setAdding] = useState(false);
  const [newEntry, setNewEntry] = useState({ type: 'experience', title: '', content: '', date: '' });

  const ENTRY_TYPES = [
    { id: 'experience', label: 'Trải nghiệm', icon: '⭐', color: 'text-hazard' },
    { id: 'repair',     label: 'Sửa chữa',    icon: '🔧', color: 'text-[#6ec070]' },
    { id: 'memory',     label: 'Kỷ niệm',     icon: '💭', color: 'text-rust' },
    { id: 'location',   label: 'Địa điểm',    icon: '📍', color: 'text-bone-2' },
    { id: 'other',      label: 'Khác',         icon: '📌', color: 'text-concrete' },
  ];

  // Mock entries để demo UI
  const entries = [
    { id: '1', type: 'memory', title: 'Lần đầu mang đến concert', content: 'Mang đến concert Đen Vâu tại Mỹ Đình. Không khí tuyệt vời, đôi giày đã chứng kiến khoảnh khắc đó.', date: '2026-03-15', icon: '💭', color: 'text-rust' },
    { id: '2', type: 'repair', title: 'Vệ sinh sole', content: 'Đem đến shop XYZ làm sạch sole và box toe. Trả về tình trạng như mới.', date: '2026-02-10', icon: '🔧', color: 'text-[#6ec070]' },
    { id: '3', type: 'location', title: 'Hà Nội → HCM', content: 'Đây là chuyến đầu tiên bay vào Nam cùng đôi giày này.', date: '2026-01-20', icon: '📍', color: 'text-bone-2' },
  ];

  const filtered = entries.filter(e => {
    const matchFilter = filter === 'all' || e.type === filter;
    const matchSearch = !search || e.title.toLowerCase().includes(search.toLowerCase()) || e.content.toLowerCase().includes(search.toLowerCase());
    return matchFilter && matchSearch;
  });

  return (
    <div className="space-y-4">
      <div className="font-mono text-[10px] text-rust tracking-[0.2em] uppercase flex items-center gap-2 before:content-[''] before:w-2 before:h-2 before:bg-rust">
        Nhật ký {objectLabel}
      </div>

      {/* Search + Filter */}
      <div className="flex gap-3 max-sm:flex-col">
        <input
          type="text"
          placeholder="🔍 Tìm kiếm trong nhật ký..."
          value={search}
          onChange={e => setSearch(e.target.value)}
          className="flex-1 bg-ink border border-line text-bone text-sm p-2 font-mono focus:outline-none focus:border-bone-2"
        />
        <select
          value={filter}
          onChange={e => setFilter(e.target.value)}
          className="bg-ink border border-line text-bone text-sm p-2 font-mono focus:outline-none"
        >
          <option value="all">Tất cả</option>
          {ENTRY_TYPES.map(t => <option key={t.id} value={t.id}>{t.label}</option>)}
        </select>
      </div>

      {/* Entry list */}
      <div className="space-y-2">
        {filtered.length === 0 ? (
          <div className="text-center py-8 text-concrete font-mono text-sm">
            {search ? `Không tìm thấy "${search}"` : 'Chưa có nhật ký nào'}
          </div>
        ) : (
          filtered.map(entry => (
            <div key={entry.id} className="border border-line p-4">
              <div className="flex items-start justify-between gap-3">
                <div className="flex items-start gap-3">
                  <span className={`text-lg flex-shrink-0 ${entry.color}`}>{entry.icon}</span>
                  <div>
                    <div className="font-mono text-[11px] font-bold text-bone mb-1">{entry.title}</div>
                    <div className="font-body text-sm text-bone-2 leading-[1.5]">{entry.content}</div>
                    <div className="font-mono text-[9px] text-concrete mt-2 tracking-[0.1em]">
                      {new Date(entry.date).toLocaleDateString('vi-VN')}
                    </div>
                  </div>
                </div>
              </div>
            </div>
          ))
        )}
      </div>

      {/* Add entry — owner only */}
      {isOwner && (
        adding ? (
          <div className="border border-rust/30 bg-rust/5 p-4 space-y-3">
            <div className="font-mono text-[9px] text-concrete tracking-[0.15em] uppercase">Thêm nhật ký mới</div>
            <div className="flex gap-2 flex-wrap">
              {ENTRY_TYPES.map(t => (
                <button key={t.id} onClick={() => setNewEntry(e => ({ ...e, type: t.id }))}
                  className={`px-3 py-1 border font-mono text-[9px] tracking-[0.1em] uppercase transition-colors
                    ${newEntry.type === t.id ? 'border-rust bg-rust/10 text-rust' : 'border-line text-concrete hover:border-bone-2'}`}
                >
                  {t.label}
                </button>
              ))}
            </div>
            <input
              type="text" placeholder="Tiêu đề..."
              value={newEntry.title}
              onChange={e => setNewEntry(n => ({ ...n, title: e.target.value }))}
              className="w-full bg-ink border border-line text-bone text-sm p-2 font-mono focus:outline-none focus:border-bone-2"
            />
            <textarea
              placeholder="Kể câu chuyện của bạn với vật phẩm này..."
              value={newEntry.content}
              onChange={e => setNewEntry(n => ({ ...n, content: e.target.value }))}
              rows={3}
              className="w-full bg-ink border border-line text-bone text-sm p-2 font-mono resize-none focus:outline-none focus:border-bone-2"
            />
            <input
              type="date" value={newEntry.date}
              onChange={e => setNewEntry(n => ({ ...n, date: e.target.value }))}
              className="bg-ink border border-line text-bone text-sm p-2 font-mono focus:outline-none"
            />
            <div className="flex gap-3">
              <button
                onClick={() => { setAdding(false); }}
                disabled={!newEntry.title || !newEntry.content}
                className="flex-1 bg-rust text-ink py-2 font-mono text-[10px] font-bold tracking-[0.14em] uppercase hover:bg-bone transition-colors disabled:opacity-40"
              >
                Lưu nhật ký
              </button>
              <button onClick={() => setAdding(false)}
                className="border border-line text-concrete px-4 py-2 font-mono text-[10px] uppercase hover:text-bone transition-colors"
              >
                Huỷ
              </button>
            </div>
          </div>
        ) : (
          <button
            onClick={() => setAdding(true)}
            className="w-full border border-dashed border-line text-bone-2 py-3 font-mono text-[10px] tracking-[0.14em] uppercase hover:border-rust hover:text-rust transition-colors"
          >
            + Thêm nhật ký mới
          </button>
        )
      )}
    </div>
  );
}

// ── Helper component ──────────────────────────────────────────

function ActionBtn({
  label, icon, active, danger, loading, onClick,
}: {
  label: string; icon: string; active?: boolean;
  danger?: boolean; loading?: boolean;
  onClick: () => void;
}) {
  return (
    <button
      onClick={onClick}
      disabled={loading}
      className={`flex items-center gap-2 px-4 py-2 border font-mono text-[10px] tracking-[0.14em] uppercase transition-colors disabled:opacity-40
        ${active
          ? 'border-rust bg-rust/10 text-rust'
          : danger
          ? 'border-rust/50 text-rust/70 hover:border-rust hover:text-rust'
          : 'border-line text-bone-2 hover:border-rust hover:text-rust'
        }`}
    >
      <span style={{ fontSize: 14 }}>{icon}</span>
      {loading ? '...' : label}
    </button>
  );
}
